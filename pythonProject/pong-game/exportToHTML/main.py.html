main.py 
from turtle import Screen, Turtle
from paddle import Paddle
from ball import Ball
import time # To slow down the ball
from scoreboard import ScoreBoard

screen = Screen()
screen.setup(800, 600)
screen.title("iNNK.PongGame")
screen.bgcolor("black")
screen.tracer(0) # it enables od disable animation

r_paddle = Paddle((350, 0))
l_paddle = Paddle((-350, 0))

ball = Ball()

scoreboard = ScoreBoard()

screen.listen()
screen.onkey(r_paddle.go_up, "Up")
screen.onkey(r_paddle.go_down, "Down")
screen.onkey(l_paddle.go_up, "w")
screen.onkey(l_paddle.go_down, "s")

game_is_on = True
while game_is_on:
    time.sleep(ball.move_speed) # To slow down the ball
    screen.update() # It updates a screen when upper code is executed
    ball.move()

    # Detecting collision with wall i.e. top and bottom
    if ball.ycor() > 280 or ball.ycor() < -280:
        # Needs to bounce
        ball.bounce_y()

    # Detecting collision with paddles
    if ball.distance(r_paddle) < 50 and ball.xcor() > 320 or ball.distance(l_paddle) < 50 and ball.xcor() < -320:
        ball.bounce_x()

    # R_Paddle miss the ball
    if ball.xcor() > 380:
        ball.reset_position()
        scoreboard.l_point()

    # L_Paddle miss the ball
    if ball.xcor() < -380:
        ball.reset_position()
        scoreboard.r_point()

















screen.exitonclick()
